   int maxSubarraySum(int a[], int n){
        
        // Your code here
        int csum=a[0],msum=a[0];
        
        for(int i=1;i<n;i++){
            
            if(csum>=0){
                csum+=a[i];
            }
            else{
                csum=a[i];
            }
            msum = max(msum,csum);
        }
        
        return msum;
        
    }
